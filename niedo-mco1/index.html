⚡ activity6 / script.js
document.addEventListener("DOMContentLoaded", function() {
  console.log("JavaScript Loaded!");

  const formSubmit  = document.querySelector("form");
  const dataContent = document.querySelector(".dataContent");

  const name    = document.getElementById("name");
  const email   = document.getElementById("email");
  const contact = document.getElementById("contact");
  const date    = document.getElementById("date");
  const service = document.getElementById("service");
  const notes   = document.getElementById("notes");

  formSubmit.addEventListener("submit", function(event) {
    event.preventDefault();

    dataContent.innerHTML = `
      <h1 class="titleContent">Tiptop Portraiture Studio</h1>
      <p><b>Client Name:</b> ${name.value}</p>
      <p><b>Email:</b> ${email.value}</p>
      <p><b>Contact Number:</b> ${contact.value}</p>
      <p><b>Reservation Date:</b> ${date.value}</p>
      <p><b>Portrait Service:</b> ${service.value}</p>
      <p><b>Notes:</b> ${notes.value}</p>
    `;

    name.value    = "";
    email.value   = "";
    contact.value = "";
    date.value    = "";
    service.value = "";
    notes.value   = "";
  });

});
